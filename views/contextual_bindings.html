inversify.config.ts

/// <reference path="../typings/inversify/inversify.d.ts" />
/// <reference path="./interfaces.d.ts"/>

import { Kernel, Binding } from "inversify";
import { Katana } from "./wespons/katana";
import { Shuriken } from "./wespons/shuriken";

var kernel = new Kernel();

var katanaBinding = new Binding("IWeapon").to(Katana).when((target) => { return target.name.equal("main"); });
var shurikenBinding = new Binding("IWeapon").to(Shuriken).when((target) => { return target.name.equal("secondary"); });

kernel.bind(katanaBinding);
kernel.bind(shurikenBinding);
